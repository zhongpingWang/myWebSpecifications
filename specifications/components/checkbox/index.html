复选按钮的样式

复选按钮

实例：
html:

<span class="groupSelect"> <label class="btnCk">篮球 </span>

js:

$(".groupSelect").myRadioCk({
	click:function(){
		 alert($(this).find(".selected").length?"选中":"未选中");
	}
});

复选按钮禁用

实例：
html:

<span class="groupSelect"> <label class="btnCk disable">篮球 </span>

js:

$(".groupSelect").myRadioCk({
    click:function(){
         alert($(this).find(".selected").length?"选中":"未选中");
    }
});
