单选按钮的样式

单选按钮

实例：
html:

<span class="groupRadio"> <label class="btnRadio">篮球 <label class="btnRadio">足球 <label class="btnRadio">球球 </span>

js:

$(".groupRadio").myRadioCk({
	click:function(){
		alert($(this).find(".selected").text());
	}
});

单选按钮禁用

实例：
html:

<span class="groupRadio"> <label class="btnRadio disable">篮球 <label class="btnRadio">足球 <label class="btnRadio">球球 </span>

js:

$(".groupRadio").myRadioCk({
    click:function(){
        alert($(this).find(".selected").text());
    }
});
